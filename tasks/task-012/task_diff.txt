+function parseIntegerList(raw, res) {
+  if (typeof raw !== 'string' || raw.trim() === '') {
+    res.status(400).json({ error: 'nums required' });
+    return null;
+  }
+
+  const tokens = raw.split(',');
+  const nums = [];
+
+  for (const t of tokens) {
+    const s = String(t).trim();
+    if (s === '' || !/^[-]?\d+$/.test(s)) {
+      res.status(400).json({ error: 'invalid nums' });
+      return null;
+    }
+    nums.push(parseInt(s, 10));
+  }
+
+  if (!nums.length) {
+    res.status(400).json({ error: 'nums required' });
+    return null;
+  }
+
+  return nums;
+}
